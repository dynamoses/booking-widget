<script>
import * as searchConstants from '../js/searchConstants';
import axios from 'axios';
import VueHotelDatepicker from '@northwalker/vue-hotel-datepicker'
import AddGuestsPicker from './AddGuestsPicker.vue'
import VueMonthlyPicker from 'vue-monthly-picker'
 
export default {
  components: {
    'vue-hotel-datepicker': VueHotelDatepicker,
    'add-guests-picker': AddGuestsPicker,
    'month-picker': VueMonthlyPicker
  },
  data() {
    return {
      selectedDates: {},
      guestPickerActive: false,
      selectedMonth: ''
    }
  },
  computed: {
      selectedAdults() {
          return this.$store.state.userSelectedData.adults;
      },
      selectedChildren() {
          return this.$store.state.userSelectedData.children;
      },
      store() {
        return this.$store;
      },
      trips() {
        return this.$store.state.datesRates.data;
      },  
      selectedJumpToMonth() {
        return this.selectedMonth._i;
      }
 
  },
  methods: {
    getDatesRates() {
      axios.get(searchConstants.tripApi + searchConstants.region + '/' + searchConstants.brand + '/' + searchConstants.tourId + '/' + searchConstants.optionId).then((data) => {
        this.$store.dispatch('updateDatesRates', data)
      }) 
    },
    getDay(val) {
      return this.$store.state.days[new Date(val).getUTCDay()];
    },
    getDate(val) {
      return ('0' + new Date(val).getUTCDate()).slice(-2);
    },
    getMonth(val) {
      return this.$store.state.months[new Date(val).getUTCMonth()];
    },
    updateSelectedDates(val) {
      this.selectedDates = val;
    },
    openGuestPicker() {
      this.guestPickerActive = true;
    },
    closeGuestPicker() {
      this.guestPickerActive = false;
    }
  },
  created() {
    this.getDatesRates();
  }
}
</script>
