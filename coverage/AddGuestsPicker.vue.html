<template>
  <div v-if="guestpickeractive" class="absolute guest-picker bg-white border brder-solid border-gray-300 w-full p-2">
      <div class="guest-picker-header text-center border-b-1">
          <div class="text-xl">How many people are travelling?</div>
          <div class="text-sm">You can add a maximum of 8 people</div>
      </div>
      <div class="pickers p-4">
          <div class="pickers__adults border-b-1 border-gray-300 flex justify-between">
              <span class="text-left">
                  <div class="adults">Adults</div>
                  <div class="age-range text-xs text-gray-700">18+ years</div>
              </span>
              <span class="counter flex">
                    <span :class="{'pointer-events-none cursor-not-allowed' : selectedAdults == 0}" class="cursor-pointer" @click="updateSelectedAdults(-1)">
                      <i class="fas fa-minus-circle" :class="{'text-gray-500' : selectedAdults == 0}"></i>
                    </span>
                  <span class="count px-3" v-text="selectedAdults"></span>
                  <span class="cursor-pointer" @click="updateSelectedAdults(1)"><i class="fas fa-plus-circle"></i></span>
              </span>
          </div>
          <div class="pickers__children flex justify-between">
              <span class="text-left">
                  <div class="adults">Children</div>
                  <div class="age-range text-xs text-gray-700">5-17 years</div>
              </span>
              <span class="counter flex">
                    <span :class="{'pointer-events-none cursor-not-allowed' : selectedChildren == 0}" class="cursor-pointer" @click="updateSelectedChildren(-1)">
                      <i class="fas fa-minus-circle" :class="{'text-gray-500' : selectedChildren == 0}"></i>
                    </span>
                  <span class="count px-3" v-text="selectedChildren"></span>
                  <span class="cursor-pointer" @click="updateSelectedChildren(1)"><i class="fas fa-plus-circle"></i></span>
              </span>
          </div>
          <div class="terms text-xs text-left p-2 mt-2 bg-gray-200 rounded-sm"><strong>Please note: </strong>Children must be between 5-17 years old at the time of travel for this trip.</div>
      </div>
      <div class="guest-picker-footer flex justify-center">
          <button class="text-sm text-white bg-pink-600 rounded-full py-2 px-3 cursor-pointer hover:bg-pink-800" @click="$emit('close-guest-picker')">Update guests</button>
      </div>
      
    <div @click="$emit('close-guest-picker')" class="guest-picker-close cursor-pointer"><i class="fas fa-times"></i></div>
  </div>
</template>
 
<script>
export default {
    props: {
        guestpickeractive: {
            type: Boolean
        }
    },
    data() {
        return {
        }
    },
    computed: {
        selectedAdults() {
            return this.$store.state.userSelectedData.adults;
        },
        selectedChildren() {
            return this.$store.state.userSelectedData.children;
        }
    },
    methods: {
        updateSelectedAdults(val) {
            this.$store.dispatch('updateSelectedAdults', val)
        },
        updateSelectedChildren(val) {
            this.$store.dispatch('updateSelectedChildren', val)
        }
    }
}
</script>
